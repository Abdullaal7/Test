pipeline {
    agent { label 'test_con' }

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/abdullaal7/TestB.git', branch: 'test'
            }
        }

        stage('Build Application') {
            steps {
                script {
                    // Assuming you're using Maven to build the Java project
                    bat 'mvn clean package'
                }
            }
        }

        stage('Build and Push with Docker Compose') {
            steps {
                script {
                    def imageName = 'abdullaal77/player-app:latest'

                    // Build the Docker image using Docker Compose
                    bat "docker-compose build"

                    // Tag the correct image name generated by Docker Compose
                    bat "docker tag testb-player ${imageName}"

                    // Log in to the Docker registry
                    bat "docker login -u abdullaal77 -p #\$AaPwD56"

                    // Push the image to the private repository
                    bat "docker push ${imageName}"
                }
            }
        }

        stage('Deploy with Docker Compose') {
            steps {
                script {
                    // Pull and run the Docker image using Docker Compose
                    bat "docker-compose pull"
                    bat "docker-compose up -d"
                }
            }
        }
    }
}
